df['통화요금/가입일'] = (df['주간통화요금'] + df['저녁통화요금'] + df['밤통화요금'] ) / df['가입일'] 

df['총통화시간'] = df['주간통화시간'] + df['저녁통화시간'] + df['밤통화시간'] 

df['총통화횟수'] = df['주간통화횟수'] + df['저녁통화횟수'] + df['밤통화횟수'] 

df['총통화요금'] = df['주간통화요금'] + df['저녁통화요금'] + df['밤통화요금'] 

df['상담전화건수/통화시간'] = df['상담전화건수'] / (df['주간통화시간'] + df['저녁통화시간'] + df['밤통화시간'] ) 

df['Max통화시간'] = df[['주간통화시간','저녁통화시간', '밤통화시간']].max(axis=1)

df['Min통화시간'] = df[['주간통화시간','저녁통화시간', '밤통화시간']].min(axis=1)

df['Diff통화시간'] = df['Max통화시간'] - df['Min통화시간']

위의 알고리즘을 통한 오버샘플링 방법은 클래스의 비중이 비슷해질때까지 계속 데이터를 생성한다.¶
때문에 심하게 불균형한 데이터의 비중을 맞추기 위해 과하게 데이터를 생성하다보니 모델 성능이 오히려 떨어지는 결과가 생긴게 아닐까 싶다.¶
따라서 데이터를 생성하지 않고 소수의 클래스에 가중치를 더 많이 주는 방법을 사용해보고자 했다.¶

cost-sensitive learning 방법¶

weights={0:1.0, 1:8.45}

임의로가중치를 어떻게 주냐에 따라 정확도가 달라지는데, 소숫점 단위로 다양하게 시도해본 결과 1 : 8.45 로 했을때 모델성능이 향상이 일어났다.¶
1:8.45로 했을때 0.8921 -> 0.893 으로 성능 향상¶
